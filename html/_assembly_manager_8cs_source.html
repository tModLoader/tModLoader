226  internal static List<Mod> InstantiateMods(List<LocalMod> modsToLoad) {
227  var modList = new List<LoadedMod>();
228  foreach (var loading in modsToLoad) {
229  if (!loadedMods.TryGetValue(loading.Name, out LoadedMod mod))
230  mod = loadedMods[loading.Name] = new LoadedMod();
231 
232  mod.SetMod(loading);
233  modList.Add(mod);
234  }
235 
236  RecalculateReferences();
237 
238  if (Debugger.IsAttached) {
239  ModLoader.isModder = true;
240  foreach (var mod in modList.Where(mod => mod.properties.editAndContinue && mod.CanEaC))
241  mod.EnableEaC();
242  }
243  if (ModLoader.alwaysLogExceptions)
244  ModCompile.ActivateExceptionReporting();
245 
246  try
247  {
248  //load all the assemblies in parallel.
249  int i = 0;
250  Parallel.ForEach(modList, mod =>
251  {
252  Interface.loadMods.SetProgressCompatibility(mod.Name, i++, modsToLoad.Count);
253  mod.LoadAssemblies();
254  });
255 
256  //Assemblies must be loaded before any instantiation occurs to satisfy dependencies
257  return modList.Select(Instantiate).ToList();
258  }
259  catch (AggregateException ae)
260  {
261  ErrorLogger.LogMulti(ae.InnerExceptions.Select(e => new Action(() => {
262  var mod = modList.Single(m => m.Name == (string)e.Data["mod"]);
263  ModLoader.DisableMod(mod.Name);
264  ErrorLogger.LogLoadingError(mod.Name, mod.modFile.tModLoaderVersion, e);
265  })));
266  return null;
267  }
268  catch (Exception e)
