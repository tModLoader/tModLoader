327  public static string HumanTimeSpanString(DateTime yourDate)
328  {
329  var ts = new TimeSpan(DateTime.UtcNow.Ticks - yourDate.Ticks);
330  double delta = Math.Abs(ts.TotalSeconds);
331 
332  if (delta < 1 * MINUTE)
333  return ts.Seconds == 1 ? "1 second ago" : ts.Seconds + " seconds ago";
334 
335  if (delta < 2 * MINUTE)
336  return "1 minute ago";
337 
338  if (delta < 45 * MINUTE)
339  return ts.Minutes + " minutes ago";
340 
341  if (delta < 90 * MINUTE)
342  return "1 hour ago";
343 
344  if (delta < 24 * HOUR)
345  return ts.Hours + " hours ago";
346 
347  if (delta < 48 * HOUR)
348  return "1 day ago";
349 
350  if (delta < 30 * DAY)
351  return ts.Days + " days ago";
352 
353  if (delta < 12 * MONTH)
354  {
355  int months = Convert.ToInt32(Math.Floor((double)ts.Days / 30));
356  return months <= 1 ? "1 month ago" : months + " mnths ago";
357  }
358  else
359  {
360  int years = Convert.ToInt32(Math.Floor((double)ts.Days / 365));
361  return years <= 1 ? "1 year ago" : years + " years ago";
362  }
363  }
