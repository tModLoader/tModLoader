235  private void PopulateModBrowser()
236  {
237  loading = true;
238  SetHeading("Mod Browser");
239  modListAll.Clear();
240  try
241  {
242  ServicePointManager.Expect100Continue = false;
243  string url = "http://javid.ddns.net/tModLoader/listmods.php";
244  var values = new NameValueCollection
245  {
246  { "modloaderversion", ModLoader.versionedName },
247  };
248  using (WebClient client = new WebClient())
249  {
250  ServicePointManager.ServerCertificateValidationCallback = new RemoteCertificateValidationCallback((sender, certificate, chain, policyErrors) => { return true; });
251  client.UploadValuesCompleted += new UploadValuesCompletedEventHandler(UploadComplete);
252  client.UploadValuesAsync(new Uri(url), "POST", values);
253  }
254  }
255  catch (WebException e)
256  {
257  if (e.Status == WebExceptionStatus.Timeout)
258  {
259  SetHeading("Mod Browser OFFLINE (Busy)");
260  return;
261  }
262  if (e.Status == WebExceptionStatus.ProtocolError)
263  {
264  var resp = (HttpWebResponse)e.Response;
265  if (resp.StatusCode == HttpStatusCode.NotFound)
266  {
267  SetHeading("Mod Browser OFFLINE (404)");
268  return;
269  }
270  SetHeading("Mod Browser OFFLINE..");
271  return;
272  }
273  }
274  catch (Exception e)
275  {
276  ErrorLogger.LogModBrowserException(e);
277  return;
278  }

319  private void PopulateFromJSON(List<BuildProperties> modBuildProperties, string json)
320  {
321  try
322  {
323  JObject jsonObject = JObject.Parse(json);
324  JObject updateObject = (JObject)jsonObject["update"];
325  if(updateObject != null)
326  {
327  updateAvailable = true;
328  updateText = (string)updateObject["message"];
329  updateURL = (string)updateObject["url"];
330  }
331  JArray modlist = (JArray)jsonObject["modlist"];
332  foreach (JObject mod in modlist.Children<JObject>())
333  {
334  string displayname = (string)mod["displayname"];
335  string name = (string)mod["name"];
336  string version = (string)mod["version"];
337  string author = (string)mod["author"];
338  string download = (string)mod["displayname"];
339  int downloads = (int)mod["downloads"];
340  string timeStamp = (string)mod["updateTimeStamp"];
341  bool exists = false;
342  bool update = false;
343  foreach (BuildProperties bp in modBuildProperties)
344  {
345  if (bp.displayName.Equals(displayname))
346  {
347  exists = true;
348  if (!bp.version.Equals(new Version(version.Substring(1))))
349  {
350  update = true;
351  }
352  }
353  }
354  UIModDownloadItem modItem = new UIModDownloadItem(displayname, name, version, author, download, downloads, timeStamp, update, exists);
355  modListAll._items.Add(modItem); //add directly to the underlying, SortList will repopulate it anyway
356  }
357  SortList();
358  }
359  catch (Exception e)
360  {
361  ErrorLogger.LogModBrowserException(e);
362  return;
363  }
